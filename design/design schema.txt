# Community stores information about anonymous posts.
# We store in database their true identity but they appear to be 'anonymous' in application.
# In this way, family can track their uploads to community and modify them.
Community {
    "posts": [ObjectId], # ids of posts
    # something more
}

# Posts in community will be visible to other families.
Post {
    "title": String,
    "family": ObjectId,
    "items": ObjectId, # this is artifact(s) a family shares.
    "datePosted": Date,
    "description": String,
    "replies": [{
      "alias": String, # e.g. Elephant, Lion etc.
      "dateReplied": Date,
      "content": String
    }]
}

# Registry
Registry {
    "item": [ObjectId],
    "description": String
}

# Item is the item uploaded by someone in some family.
Item {
    "name": String,
    "dateAdded": Date,
    "textDesc": String,
    "voiceDesc": String, #id in Amazon S3
    "image": image, # all images are String from Amazon S3
    # add more?
}

FamilyMember {
    "avatar": image,
    "firstName": String,
    "middleName": String,
    "lastName": String,
    "sex": Enum, # not boolean since more than 2 sex types?
    "registry": ObjectId, # id of registry
}


# UUID is not needed since MongoDB will automatically assign an ID to each entry.
FamilyAccount {
    "name": String
    "profile": [ObjectId], # ids of profile
    "password": String, # preferrably hashed
    "dateCreated": Date,
    "email": String,
    # to be added: achievement etc.
}
